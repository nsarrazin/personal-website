Graph generation¶

We use the networkx python library to create a graph, mathematical structure that represents each sensor as a node, and the distance between the neighboring nodes is represented as weighted edges, where the weight of the edge is the euclidean distance between them.

In [9]:
import networkx as nx

def createGraph(timeframe):
    dic = getFrameDict(timeframe)
    structure = nx.Graph(name="Sensor graph at timeframe " +str(timeframe))
    for i in range(1,57):
        structure.add_node(i, pos=dic[str(i)])

    for i in range(1,57):
        for j in getNeighbours(i):
            iCoord = dic[str(i)]
            jCoord = dic[str(j)]
            distance = np.linalg.norm(np.array(iCoord)-np.array(jCoord))
            structure.add_edge(i,j, dist=distance)


    return structure

import functools

def createAverageGraph(PostureList):
    """
    Inputs a list of timeframes for a specific posture
    
    Outputs a networkx graph representing the average posture sensor graph
    """
    listDic = []
    for tf in PostureList:
        graph = createGraph(tf)
        dic = nx.get_node_attributes(graph, "pos")
        listDic.append(dic)
    
    # beware, this is pure dark magic, but it works ¯\_(ツ)_/¯ 
    # thanks to @ojdo https://stackoverflow.com/a/19654913
    
    dicList = {k:[ listDic[j][k] for j in range(len(listDic)) ] for k in listDic[0].keys()}
    T = {k:( functools.reduce(np.add, v)/len(v) ) for k,v in dicList.items()}  
    avgGraph = nx.Graph()
    
    for i in range(1,57):
        avgGraph.add_node(i, pos = T[i])

    for i in range(1,57):
        for j in getNeighbours(i):
            iCoord = T[i]
            jCoord = T[j]
            distance = np.linalg.norm(np.array(iCoord)-np.array(jCoord))
            avgGraph.add_edge(i,j, dist=distance)

    return avgGraph

Delta-graphs generation¶

We will now generate delta-graphs, where the edges no longer represent absolute distance between points but rather, the change in distance between a base graph and a subject graph.

In [19]:
def createDeltaGraph(baseGraph, subjectGraph):
    deltaGraph = nx.Graph()
    
    baseGraphDist = nx.get_edge_attributes(baseGraph, 'dist')
    subjGraphDist = nx.get_edge_attributes(subjectGraph, 'dist')
    
    for i in range(1, 57):
        deltaGraph.add_node(i)
    
    for key in baseGraphDist.keys():
        relativeDistance = (subjGraphDist[key]-baseGraphDist[key])/(baseGraphDist[key])
        deltaGraph.add_edge(*key, rDist = relativeDistance)
    
    return deltaGraph

def plotSensorMap(graph, title):
    def getColor(dist):
        red=224 - (224-49)*dist
        green =243 - (243-54)*dist
        blue =248- (248-149)*dist
        return red,green,blue
    xNodes = []
    yNodes = []
    
    for i in graph.nodes():
        y, x = getPos(i)
        xNodes.append(x)
        yNodes.append(-y)
        
    rDist = nx.get_edge_attributes(graph, 'dist')
    
    mDist = max(rDist.values())
    maxThick = 7
    data = []
    
    for i in graph.edges():
        dist = rDist[i]
        t = (1-np.abs((dist-mDist)/mDist))
        r,g,b = getColor(t**2)
        t = (t**2)*maxThick #we use a square function to better display the difference between small and big distance
        edge_trace = go.Scatter(
                                x=[],
                                y=[],
                                line=go.Line(width=t, color="rgb({},{},{})".format(r,g,b)),
                                hoverinfo='none',
                                mode='lines')

        y0, x0 = getPos(i[0])
        y1, x1 = getPos(i[1])
        edge_trace['x'] += [x0, x1, None]
        edge_trace['y'] += [-y0, -y1, None]
        data.append(edge_trace)

    trace0 = go.Scatter(
        x = xNodes,
        y = yNodes,
        name = 'Sensor nodes',
        mode = 'markers',
        marker = dict(
            size = 10,
            color = 'rgba(100, 100, 100, .8)',
            line = dict(
                width = 2,
                color = 'rgba(0, 0, 0, .8)'
            )))
    
    data.append(trace0)
    layout = go.Layout(
        hovermode= 'closest',
        showlegend=False,
        title=title,
        xaxis=dict(
            autorange=True,
            showgrid=False,
            zeroline=False,
            showline=False,
            autotick=True,
            ticks='',
            showticklabels=False
        ),

            autorange=True,

            ticks='',
            showticklabels=False
        ))

Optimal value search¶

Recap¶

We are in possession of graphs representing our sensor set, where each node on the graph is a sensor, with a position attribute and each edge - linking neighboring nodes - has a weight representing the distance between the nodes it links.

We created average models of each postures to allow easier generalization. This can however be revisited later and maybe replaced by a more accurate method.

We also created what we called "delta-graphs" where the edges no longer represent the distance between sensors, but the change in distance based on two different postures

We also figured out a way to find the pair of sensor with the maximum change in distance.
